! Simple domain blocking
||example.com^

! Subdomain blocking (inherits domain block)
||sub.example.com^

! Wildcard blocking in URL (images)
|http://*ads.example.*^$image

! Beginning-of-URL anchor
|http://ads.example.com^banner.gif

! End-of-URL anchor
banner.example.com|

! Separator anchor (^ matches start/end, slash, & or end-of-URL)
||example.com^/path/to/asset.js

! Exception rule (whitelisting)
@@||example.com^$third-party

! Element hiding (CSS selector)
example.com###ad_banner

! Element hiding with pseudo-class
example.com##.ad-slot:nth-of-type(2)

! Cosmetic exception (CSS selector exception)
example.com#@#.ad-slot

! CSS injection (style override)
example.com##body:style(position:fixed!important;background:red!important;)

! Regular expression rule (simple)
/banner_[0-9]+\.png$/

! Regular expression with resource type
/https:\/\/(ads|track)\.example\.com\/.*\.js$/$script

! Blocking with single option
||ads.example.com^$script

! Blocking with multiple options and domains
||ads.example.com^$script,image,third-party,domain=example.com|~forum.example.com

! Blocking with domain-only option
||ads.only.com^$domain=example.com

! Domain exception within options
||ads.only.com^$domain=example.com|~sub.example.com

! Unknown/invalid option must be skipped
||ads.example.com^$invalid_option,stylesheet

! Leading/trailing whitespace around rule
   ||whitespace.example.com^

! Universal wildcard rule (should be valid but match all)
/.*$/

! Single literal asterisk rule (literal, not wildcard)
|*|

||xn--exmple-cua.com^

! Blocking on custom port
||example.com:6969^

! Query-string option
||track.example.com^$querystring=ref

||CaseSensitive.COM^$match-case

||example.com^$removeparam=/sessid|token/

||fonts.example.com^$font,stylesheet

||example.com^$~third-party

|^|
